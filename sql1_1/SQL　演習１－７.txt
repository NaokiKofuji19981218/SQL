問１テーブルbooksから価格順に本を並べ、価格が２５００円以上３５００円以下の情報を出力する

//SELECT命令を出力する。テーブルからデータを取り出す
SELECT　
　//titleの列を取り出す。
  title,
　//priceの列を取り出す。
  price
//何のテーブルからデータを取り出すか
FROM
  //参照するテーブル名の入力
  books
//条件を決める。　　
WHERE
  //条件の入力
  price BETWEEN 2500 AND 3500 
//情報を並べる順序を決める。　
ORDER BY
  //条件の入力
  price ASC    
;




問２テーブルusrから姓と名の順に神奈川県、東京都、千葉県に住んでいる人の情報を並べる。

//SELECT命令を出力する。テーブルからデータを取り出す
SELECT　
　//l_nameの列を取り出す。
  l_name,
　//f_nameの列を取り出す。
  f_name,
　//prefectureの列を取り出す。
  prefecture
//何のテーブルからデータを取り出すか
FROM
  //参照するテーブル名の入力
  usr
//条件を決める。　
WHERE
  //条件の入力
    prefecture = '東京都' 
  //条件の追加
　AND
 　 //条件の入力
    prefecture = '千葉県'
  //条件の追加
　AND
    //条件の入力
    prefecture = '神奈川県' 
//情報を並べる順序を決める。　
ORDER BY
  //条件の入力
  l_name ASC  
  //条件の入力  
  f_name ASC 
;




問３テーブルrentalから貸出日が２０１２年１２月１日より前の情報を取り出す。


//SELECT命令を出力する。テーブルからデータを取り出す
SELECT　
　//全ての列を取り出す。
  *
//何のテーブルからデータを取り出すか
FROM
  //参照するテーブル名の入力
  rental
//条件を決める。
WHERE
    //条件の入力
    day = 0 
  //条件の追加
　AND
    //条件の入力
    date < '2012_12_1'
//情報を並べる順序を決める。
ORDER BY
  //条件の入力
  date DESC     
;

  *
FROM
  access_log 
WHERE
  access_date IN ('2013-01-01','2013-01-31')
ORDER BY
  referer,ip_address DESC
