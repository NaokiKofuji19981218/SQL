問３テーブルrentalから未返却かつ貸出日が古いものから順に出力する

//SELECT命令を出力する。テーブルからデータを取り出す
SELECT　
　//user_idの列を取り出す。　
  user_id,
　//isbnの列を取り出す。　
  isbn,
　//rental_dateの列を取り出す。
  rental_date
//何のテーブルからデータを取り出すか
FROM  
  //参照するテーブル名の入力
  rental
//条件を決める。
WHERE
  //条件の入力
  date = '0'
//情報を並べる順序を決める。
ORDER BY
  //条件の入力
  rental_date ASC 
//並べる件数を決める。
LIMIT
  //条件の入力
  5
;
