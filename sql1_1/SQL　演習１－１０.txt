問１テーブルproductから安い順に商品名と価格を出力する

//SELECT命令を出力する。テーブルからデータを取り出す
SELECT　
　//nameの列を取り出し、商品名と名前を付ける。
  name AS '商品名',
　//priceの列を取り出し、価格と名前を付ける。
  price AS '価格',
//何のテーブルからデータを取り出すか
FROM
  //参照するテーブル名の入力
  product
//情報を並べる順序を決める。
ORDER BY
　//条件の入力
  price ASC 
;




問２テーブルquestから都道府県ごとの回答者の平均年齢を出力する。

//SELECT命令を出力する。テーブルからデータを取り出す
SELECT　
　//prefectureの列を取り出し、都道府県と名前を付ける。　
  prefecture　AS 都道府県,
　//sexの列を取り出し、平均年齢と名前を付ける。　
  AGE(age) AS 平均年齢
//何のテーブルからデータを取り出すか
FROM
  //参照するテーブル名の入力
  quest
//情報を並べる順序を決める。
GROUP BY
　//条件の入力
  prefecture, 
;
