問４テーブルemployeeから氏がアで始まる情報を取り出す。

//SELECT命令を出力する。テーブルからデータを取り出す。

　//l_nameの列を取り出す
  l_name,
　//f_nameの列を取り出す
  f_name
//何のテーブルからデータを取り出すか
FROM
  //参照するテーブル名の入力
  employee
