//列を追加するテーブルを読み込む命令文

　//テーブルの指定
  books
//定義を変える列の指定
MODIFY
  publish VARCHAR(100) NULL
;

問4以下のコードの誤り２点を答える。

ALTER TABLE
  access_log
MODIFY with
  referer,VARCHAR(200),NULL
;


・MODIFY with　ではなく、　MODIFYが正しい。

・referer,VARCHAR(200),NULL ではなく、referer　VARCHAR(200)　NULLが正しい。
