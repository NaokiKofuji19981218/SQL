問１社員テーブルとタイムカードテーブルから氏名と２０１２年１２月の勤務時間を、
　　社員コード、日付について昇順で出力する。


//SELECT命令を出力する。テーブルからデータを取り出す
SELECT
　//e.l_nameの列を取り出す。
  e.l_name,
　//e.f_nameの列を取り出す。
  e.f_name,
　//t.r_dateの列を取り出す。
  t.r_date,
　//timeの列を取り出す。
  t.work_time
//何のテーブルからデータを取り出すか
FROM
　//参照するテーブル名の入力
  employee AS e
どのテーブル結合するか決める。
INNER JOIN
  //結合するテーブル名の入力
  time_card AS t
//結合するためのコード
ON
//結合する。
  e.employee = t.time_card
//条件を決める。
WHERE
　//条件の入力
  t.r_date BETWEEN '2012-12-01' AND  '2012-12-31'   
//情報を並べる順序を決める。
ORDER BY
　//条件の入力
  e.s_id ASC
　//条件の入力
  t.r_date ASC
;
