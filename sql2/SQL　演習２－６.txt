問１アンケート回答テーブルと社員テーブルから女性の氏名だけを、
　　氏・名昇順で出力する。


//SELECT命令を出力する。テーブルからデータを取り出す
SELECT
　//e.l_nameの列を取り出す。
  q.name_kana,
//何のテーブルからデータを取り出すか
FROM
　//参照するテーブル名の入力
  quest AS q

//条件を決める
WHERE
//テーブルを統合する。
UNION ALL
  //統合に用いるテーブルから情報を抽出する。
  SELECT
    CONCAT(e.l_name_kana,
           e.f_name_kana) 
  FROM
    employee AS e
  WHERE
    e.sex = 2

氏と名の昇順で重複も加えて表示する。
ORDER BY
  1 ASC
;

問３ユーザーテーブルとアンケート回答テーブルからそれぞれ千葉県に住んでいる人間だけを
　　出力し、結果を統合する。


//SELECT命令を出力する。テーブルからデータを取り出す
SELECT
　//氏と名をつなげて表示するようにする。
  CONCAT(u.l_name_kana, u.f_name_kana)

//何のテーブルからデータを取り出すか
FROM
　//参照するテーブル名の入力
  usr AS u

//表示の条件を決める
WHERE
  u.prefecture = '千葉県'

//テーブルを統合する。
UNION

    //条件の入力 
 　SELECT
     q.name_kana
   FROM
     quest AS q
  where
    q.prefecture = '千葉県'

ORDER BY
  1 DESC
;
