//既存のデータを削除する命令文
DELETE FROM
　//テーブルの指定
　employee
//条件を定める。
WHERE
　depart_id
//更新するデータの特定
NOT IN
　(
　　//列の特定
　　SELECT
　　　depart_id
　　//テーブルの指定

　　　depart
　)

問2　問１の内容を相関サブクエリを使って書き換える。

//既存のデータを削除する命令文
DELETE FROM
　//テーブルの指定
　employee
//条件を定める。
WHERE
//更新するデータの特定
NOT EXIST
　(
　//列の特定
　SELECT
　　depart.depart_id
　//テーブルの指定
　FROM
　　depart
//条件を定める。
WHERE
  depart.depart_id = employee.depart_id
)
;
